curl example 
curl -X POST http://localhost:8000/api/v1/tts \
  -F "text=Hello from Vox Studio." \
  -F "voice_name=noelmo-normal" \
  -F "preset=default"

curl http://localhost:8000/api/v1/jobs/{request_id}

curl -L http://localhost:8000/api/v1/jobs/{request_id}/audio \
  --output voice.mp3
